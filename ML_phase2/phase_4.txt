# sklean metrics

from sklearn.metrics import accuracy_score,precision_score,recall_score,f1_score

# true answer (what acutally happend)
y_true = [1,0,1,1,0,1,0]

# model's prediction (what it predicted)
y_pred = [1,0,1,0,0,1,1]

#evaluation
print('Accuracy: ',accuracy_score(y_true,y_pred))
print('precision:', precision_score(y_true,y_pred))
print('Recall:', recall_score(y_true,y_pred))
print('f1_score:',f1_score(y_true,y_pred))

* MAE --> mean absolute error
1. take the mistake difference 
2. remove minus sign 
3. add mistakes  
4. divide it by no. of students

MAE = (5+10+10+5)/4 = 7.5
MAE = 7.5 on average 

* MSE --> mean squared error
1. square each mistake 
2. add them 
3. divide by total no. of 

MSE = (25+100+100+25)/4 = 62.5

*RMSE --> root mean squared error


from sklearn.metrics import mean_absolute_error,mean_squared_error
import numpy as np 
 
# real scores
y_true = [90,60,80,100]
# pred scores
y_pred = [85,70,70,95]

mae = mean_absolute_error(y_true,y_pred)
mse = mean_squared_error(y_true,y_pred)
rmse = np.sqrt(mse)

print("MAE : on average off by : ",mae) # 7.5
print("MSE : squared mistake value : ",mse) # 62.5
print("RMSE : Final Realistic error : ",rmse) # 7.90569...
